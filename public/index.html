WITH impressions AS (
  SELECT
    user_id,
    surface,
    DATE_TRUNC('day', event_ts) AS d,
    COUNT(DISTINCT session_id) AS sessions,
    SUM(impression_count) AS count
  FROM hive.pin_telemetry.events
  WHERE event_ts BETWEEN DATE '2026-01-01'
                    AND DATE '2026-05-20'
    AND metric_type = 'unified_v2'
  GROUP BY 1, 2, 3
),
retry_queue AS (
  SELECT
    user_id,
    COUNT(event_id) AS reclaimed_count,
    AVG(retry_cooldown_ms) AS latency
  FROM indexeddb.client_cache.syncs
  WHERE status = 'successfully_flushed'
  GROUP BY 1
)
SELECT
  i.surface,
  COUNT(DISTINCT i.user_id) AS active_pinners,
  AVG(i.count) AS avg_impressions_per_dau,
  SUM(r.reclaimed_count) AS total_saved_events
FROM impressions i
LEFT JOIN retry_queue r USING (user_id)
GROUP BY 1
ORDER BY active_pinners DESC;

export async function handleOutageEvent(
  event: TelemetryEvent,
  cache: CacheDB
): Promise<void> {
  const hasNetwork = navigator.onLine;

  if (!hasNetwork) {
    // Save directly to local cache storage
    await cache.enqueue("events_queue", {
      ...event,
      timestamp: Date.now(),
      retryCount: 0
    });
    return;
  }

  try {
    await sendToCollector(event);
  } catch (err) {
    // Failover into queue for retry later
    await cache.enqueue("events_queue", event);
  }
}

export async function flushOfflineCaches(
  cache: CacheDB
): Promise<number> {
  const queued = await cache.getAll("events_queue");
  if (queued.length === 0) return 0;

  let reclaimed = 0;
  for (const ev of queued) {
    try {
      await sendToCollector(ev);
      await cache.delete("events_queue", ev.id);
      reclaimed++;
    } catch (err) {
      // Outage still active or rate-limited
      break;
    }
  }
  return reclaimed;
}
