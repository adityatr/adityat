-- engagement health, last 90 days
WITH active_pinners AS (
  SELECT
    user_id,
    country_code,
    DATE_TRUNC('day', event_ts) AS d,
    COUNT(DISTINCT session_id) AS sessions,
    SUM(impression_count)   AS impressions,
    SUM(close_up_count)      AS closeups,
    SUM(repin_count)         AS repins
  FROM hive.pin_engagement.daily_events
  WHERE event_ts BETWEEN DATE '2025-12-01'
                  AND DATE '2026-02-28'
    AND surface IN ('home_feed', 'related', 'search')
    AND impression_count > 0
  GROUP BY 1, 2, 3
),
cohort AS (
  SELECT
    a.user_id,
    a.country_code,
    MIN(a.d)                          AS first_active,
    APPROX_DISTINCT(a.d)              AS active_days,
    SUM(a.repins) * 1.0 / NULLIF(SUM(a.impressions), 0)
                                       AS repin_rate,
    SUM(a.closeups) * 1.0 / NULLIF(SUM(a.impressions), 0)
                                       AS closeup_rate
  FROM active_pinners a
  GROUP BY 1, 2
),
guardrails AS (
  SELECT
    country_code,
    APPROX_PERCENTILE(repin_rate,   0.5)  AS p50,
    APPROX_PERCENTILE(repin_rate,   0.95) AS p95,
    APPROX_PERCENTILE(closeup_rate, 0.5)  AS cu_p50
  FROM cohort
  WHERE active_days >= 7
  GROUP BY 1
)
SELECT
  c.country_code,
  COUNT(*)               AS users,
  AVG(c.repin_rate)       AS avg_repin_rate,
  AVG(c.closeup_rate)     AS avg_closeup_rate,
  g.p95                    AS p95_repin
FROM cohort c
JOIN guardrails g USING (country_code)
WHERE c.active_days >= 7
GROUP BY 1, g.p95
HAVING COUNT(*) >= 5000
ORDER BY users DESC
LIMIT 50;

import { ChatMessage, ChatResponse } from "./types";

const ENDPOINT = "https://openrouter.ai/api/v1/chat/completions";

export interface CompleteOptions {
  model: string;
  messages: ChatMessage[];
  temperature?: number;
  signal?: AbortSignal;
}

export async function complete(
  { model, messages, temperature = 0.2, signal }: CompleteOptions,
): Promise<ChatResponse> {
  const res = await fetch(ENDPOINT, {
    method: "POST",
    headers: {
      "Content-Type":  "application/json",
      "Authorization": `Bearer ${OPENROUTER_API_KEY}`,
      "HTTP-Referer":  "https://adityat.com",
      "X-Title":       "metrics-quality",
    },
    body: JSON.stringify({ model, messages, temperature, stream: false }),
    signal,
  });

  if (!res.ok) {
    throw new Error(`openrouter ${res.status}`);
  }

  return res.json() as Promise<ChatResponse>;
}

/** server-sent stream, line by line. small and honest. */
export async function* stream(opts: CompleteOptions) {
  const res = await fetch(ENDPOINT, {
    method: "POST",
    headers: headers(),
    body: JSON.stringify({ ...opts, stream: true }),
    signal: opts.signal,
  });

  const reader = res.body!.getReader();
  const decoder = new TextDecoder();
  let buf = "";

  while (true) {
    const { value, done } = await reader.read();
    if (done) break;
    buf += decoder.decode(value, { stream: true });

    for (const line of buf.split("\n\n")) {
      if (!line.startsWith("data: ")) continue;
      const payload = line.slice(6).trim();
      if (payload === "[DONE]") return;
      yield JSON.parse(payload);
    }
  }
}
